DATA SEGMENT
MSG1 DB 0AH,0DH,"ENTER STRING$"
MSG2 DB 0AH,0DH,"PALLINDROME$"
MSG3 DB 0AH,0DH,"NOT PALLINDROME $" 
STR1 DB DUP(0)
DATA ENDS

CODE SEGMENT
ASSUME CS:CODE DS:DATA
START:
MOV AX,DATA
MOV DS,AX
LEA DX,MSG1
MOV AH,09H
INT 21H
LEA SI,STR1
LEA DI,STR1
MOV AH,01H 

NEXT:
INT 21H
CMP AL,0DH
JE TERMINATE
MOV [DI],AL
INC DI
JMP NEXT 

TERMINATE:
MOV AL,'$'
MOV [DI],AL  

DOTHIS:
DEC DI
MOV AL,[DI]
CMP [SI],AL
JNE NOTPALLINDROME
INC SI
CMP SI,DI
JL DOTHIS

PALLINDROME:
LEA DX,MSG2
MOV AH,09H
INT 21H
JMP XX

NOTPALLINDROME:
LEA DX,MSG3
MOV AH,09H
INT 21H

XX:
MOV AH,4CH
INT 21H
CODE ENDS
END START
