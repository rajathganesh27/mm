DATA SEGMENT
MSG1 DB 0AH,0DH,"FOUND$"
MSG2 DB 0AH,0DH,"NOT FOUND$"
ARRAY DW 1234H,2345H,4567H,9876H,7654H
ARRAYLENGTH EQU 5
DATA ENDS

CODE SEGMENT
ASSUME CS:CODE DS:DATA
START:
MOV AX,DATA
MOV DS,AX
MOV CX,ARRAYLENGTH
LEA SI,ARRAY
MOV AX,9876H

NEXT:
CMP AX,[SI]
JE FOUND
INC SI   
INC SI
DEC CX
JNZ NEXT

NOTFOUND:
LEA DX,MSG2
MOV AH,09H
INT 21H
JMP EXIT

FOUND:
LEA DX,MSG1
MOV AH,09H
INT 21H    

EXIT:
MOV AH,4CH
INT 21H

CODE ENDS
END START
